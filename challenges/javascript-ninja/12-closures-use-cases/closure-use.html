For what can Closures be used?

  1. Encapsulation of data - create "private variables"
  2. Callbacks und Timer

Beispiel zu Fall 1: Encapsulation with Closures

    
        function Ninja() {
            var feints = 0;

            this.getFeints = function() {
                return feintsl
            }

            this.feint = function() {
                feints++;
            }
        }

        var ninja = new Ninja();

        ninja.feint();

        assert(ninja.getFeints() == 1, "We are able to access the internal feint count.");
        assert(ninja.feints === undefined, "And the private data is inaccessible now.");

Encapsulation with version of a constructor.

Beispiel zu Fall 1: Encapsulation without a constructor

    
        var ninja = (function() {

            var feints = 0;

            function getFeints() {
                return feints;
            }

            return  {
                getFeints: getFeints
            }
        });

        ninja().getFeints();

Callbacks und Timer
